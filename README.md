def desenhar_celula(xt, yt, cor='grey'):

    patio.color(cor)
    patio.speed(0)
    patio.up()
    patio.goto(xt, yt)
    patio.down()
    patio.begin_fill()

    for i in range(4):
        patio.forward(20)
        patio.left(90)

    patio.end_fill()
